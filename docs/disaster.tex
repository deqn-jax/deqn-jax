\begin{equation*}
	log(\textcolor{green}{\epsilon_{t}}) =  \rho_{\epsilon} log(\epsilon_{t-1})
											 + \sigma_{\epsilon}\epsilon_t^{\epsilon}
\end{equation*}

\begin{equation*}
	log(\textcolor{green}{\mu_{\Upsilon,t}}) =  \rho_{\mu_{\Upsilon}} log(\mu_{\Upsilon,t-1})
											 + \sigma_{\mu_{\Upsilon}} \epsilon_t^{\mu_{\Upsilon}}
\end{equation*}

\begin{equation*}
	log(\textcolor{green}{\mu_{z,t}}) =  \rho_{\mu_{z}} log(\mu_{z,t-1})
										+ (1-\rho_{\mu_{z}}) log(\mu_{z})
											 + \sigma_{\mu_{z}}\epsilon_t^{\mu_{z}}
\end{equation*}

	log(\textcolor{green}{g_{t}}) =  \rho_{g} log(g_{t-1})

											 + \sigma_{g}\epsilon_t^{g}
\end{equation*}

\section{Analytical Eliminations}

Four equilibrium equations are static (no expectations) and can be solved
analytically for their target variable. This reduces the system from 12 controls
and 12 equations to \textbf{8 controls and 8 equilibrium equations}, with 4
variables computed as definitions.

\subsection*{Eq.~10: Marginal cost $s_t$ (cost minimization FOC)}

From eq.~10 and the rental rate definition $r_t^k = \epsilon_t \alpha
(\mu_{z,t} h_t / k_{t-1})^{1-\alpha} s_t$, substitute into eq.~10 and solve for $s_t$:
\begin{equation*}
  \boxed{s_t = \frac{1}{\epsilon_t}
    \left(\frac{\mu_{z,t}\, h_t}{k_{t-1}}\right)^{\!\alpha}
    \frac{\tilde{w}_t}{1-\alpha}}
\end{equation*}
Depends on: $\epsilon_t, \mu_{z,t}, k_{t-1}$ (states), $h_t, \tilde{w}_t$ (remaining controls), $\alpha$ (parameter).

\subsection*{Eq.~12: Leverage $L_t$ (balance sheet identity)}

From eq.~12 ($k_t$ is already enforced by the state transition) and the
leverage definition $L_t n_t = q_t k_t$:
\begin{equation*}
  \boxed{L_t = \frac{q_t\, k_t}{n_t}}
\end{equation*}
Depends on: $q_t$ (remaining control), $k_t, n_t$ (definitions computed from other variables).

\subsection*{Eq.~11: Consumption $c_t$ (resource constraint)}

From eq.~11, isolate $c_t$:
\begin{equation*}
  \boxed{c_t = y_{z,t} - g_t - \frac{i_t}{\mu_{\Upsilon,t}}
    - \underbrace{\Theta\frac{1-\gamma}{\gamma}(n_t - w^e)}_{\text{entrepreneur consumption}}
    - \underbrace{\frac{\mu\, G(\bar\omega_t)\, R_t^k\, q_{t-1}\, k_{t-1}}{\mu_{z,t}\,\pi_t}}_{\text{monitoring cost}}}
\end{equation*}
Depends on: $g_t, \mu_{\Upsilon,t}, \mu_{z,t}, q_{t-1}, k_{t-1}$ (states),
$i_t, \pi_t$ (remaining controls), $y_{z,t}, n_t, R_t^k, G(\cdot)$ (definitions).
Clipped to $c_t \geq 10^{-4}$ to prevent NaN in the habit-formation Euler equation.

\subsection*{Eq.~8: Default threshold $\bar\omega_t$ (bank participation constraint)}

Eq.~8 implicitly defines $\bar\omega_t$ given all other current-period quantities.
Define:
\begin{equation*}
  h(\bar\omega) \;=\; \bar\omega\,(1-F(\bar\omega)) + (1-\mu)\,G(\bar\omega)
  \;=\; \Gamma(\bar\omega) - \mu\,G(\bar\omega)
\end{equation*}
and the target from the bank zero-profit condition:
\begin{equation*}
  \text{target} \;=\; \frac{L_{t-1} - 1}{L_{t-1}\, R_t^k / R_{t-1}}
\end{equation*}
Then eq.~8 reduces to $h(\bar\omega_t) = \text{target}$, solved by Newton's method:
\begin{equation*}
  \boxed{\bar\omega^{(n+1)} = \bar\omega^{(n)}
    - \frac{h(\bar\omega^{(n)}) - \text{target}}{h'(\bar\omega^{(n)})}}
  \qquad
  h'(\bar\omega) = 1 - F(\bar\omega) - \mu\,G'(\bar\omega)
\end{equation*}
Convergence: $h'(\bar\omega_{\text{ss}}) \approx 0.98$ (well-conditioned); 2 iterations
from $\bar\omega_0 = 0.488$ to machine precision. Implementation uses 10 fixed iterations
with $\bar\omega \in [0.01, 3.0]$ clipping, via \texttt{jax.lax.scan} for JIT compatibility.

\medskip
\noindent\textbf{Key dependency:} $R_t^k$ depends on $(h_t, \tilde{w}_t, q_t, \pi_t, \text{states})$
but \emph{not} on $\bar\omega_t$, so the computation order
$R_t^k \to \text{target} \to \bar\omega_t \to F, G, \Gamma \to n_t, L_t, c_t$
has no circular dependency. Autodiff works via the chain rule through the Newton steps.

\medskip
\noindent\textbf{Remaining system:} 8 controls
$(\lambda_z, i, \pi, \tilde{w}, h, F_w, F_p, q)$
and 8 equations (eq.~1--7, eq.~9). The eliminated variables $s_t$, $L_t$, $c_t$, $\bar\omega_t$
are available as definitions for dynamics and evaluation.
